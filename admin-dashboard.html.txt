        <div class="filters"><input type="text" id="search" placeholder="Search name, phone, location..." onkeyup="renderOrders()"><select id="statusFilter" onchange="renderOrders()"><option value="all">All Status</option><option value="New">New</option><option value="Completed">Completed</option></select><button class="export-btn" onclick="exportCSV()">📥 Export to Excel</button></div>

    <script>
        const PASSWORD = "admin123";
        let orders = [];
        function login() { if(document.getElementById('password').value === PASSWORD) { document.getElementById('loginScreen').style.display = 'none'; document.getElementById('dashboard').style.display = 'block'; loadOrders(); setInterval(loadOrders, 30000); } else { document.getElementById('error').style.display = 'block'; } }
        function logout() { document.getElementById('loginScreen').style.display = 'flex'; document.getElementById('dashboard').style.display = 'none'; document.getElementById('password').value = ''; }
        function loadOrders() { const stored = localStorage.getItem('ghanahelp_orders'); if(stored) orders = JSON.parse(stored); else orders = []; renderOrders(); updateStats(); }
        function updateStats() { document.getElementById('totalOrders').innerText = orders.length; document.getElementById('newOrders').innerText = orders.filter(o=>o.status==='New').length; document.getElementById('completedOrders').innerText = orders.filter(o=>o.status==='Completed').length; }
        function renderOrders() { const search = document.getElementById('search').value.toLowerCase(); const statusFilter = document.getElementById('statusFilter').value; let filtered = orders.filter(o => (o.name.toLowerCase().includes(search) || o.phone.includes(search) || o.location.toLowerCase().includes(search)) && (statusFilter === 'all' || o.status === statusFilter)); if(filtered.length === 0) { document.getElementById('ordersTable').innerHTML = '<div style="padding:40px; text-align:center;">📭 No orders found</div>'; return; }
            let html = '<table><thead><tr><th>ID</th><th>Date</th><th>Name</th><th>Phone</th><th>Location</th><th>Services</th><th>Status</th><th>Actions</th></tr></thead><tbody>';
            filtered.forEach(o => { html += `<tr><td>#${o.id}</td><td>${o.timestamp}</td><td><strong>${o.name}</strong></td><td>${o.phone}</td><td>${o.location}</td><td>${o.services}</td><td><span class="status-${o.status === 'New' ? 'new' : 'completed'}">${o.status}</span></td>
                <td><button class="view-btn small" onclick="viewOrder(${o.id})">👁️ View</button>${o.status !== 'Completed' ? `<button class="complete-btn small" onclick="markComplete(${o.id})">✅ Complete</button>` : ''}<button class="delete-btn small" onclick="deleteOrder(${o.id})">🗑️ Delete</button></td></tr>`; });
            html += '</tbody></table>'; document.getElementById('ordersTable').innerHTML = html;
        }
        function viewOrder(id) { const o = orders.find(x=>x.id===id); if(!o) return; document.getElementById('modalBody').innerHTML = `<p><strong>Order ID:</strong> #${o.id}</p><p><strong>Date:</strong> ${o.timestamp}</p><p><strong>Name:</strong> ${o.name}</p><p><strong>Phone:</strong> ${o.phone}</p><p><strong>Email:</strong> ${o.email || 'N/A'}</p><p><strong>Location:</strong> ${o.location}</p><p><strong>Services:</strong> ${o.services}</p><p><strong>Urgency:</strong> ${o.urgency || 'N/A'}</p><p><strong>Budget:</strong> ${o.budget || 'N/A'}</p><p><strong>Description:</strong> ${o.description}</p><p><strong>Status:</strong> ${o.status}</p>`; document.getElementById('modal').style.display = 'flex'; }
        function closeModal() { document.getElementById('modal').style.display = 'none'; }
        function markComplete(id) { const o = orders.find(x=>x.id===id); if(o && confirm(`Mark order #${id} as completed?`)) { o.status = 'Completed'; localStorage.setItem('ghanahelp_orders', JSON.stringify(orders)); renderOrders(); updateStats(); } }
        function deleteOrder(id) { if(confirm('Delete this order?')) { orders = orders.filter(x=>x.id!==id); localStorage.setItem('ghanahelp_orders', JSON.stringify(orders)); renderOrders(); updateStats(); } }
        function exportCSV() { let csv = "Order ID,Date,Name,Phone,Email,Location,Services,Urgency,Budget,Description,Status\n"; orders.forEach(o => { csv += `"${o.id}","${o.timestamp}","${o.name}","${o.phone}","${o.email || ''}","${o.location}","${o.services}","${o.urgency || ''}","${o.budget || ''}","${o.description.replace(/"/g, '""')}","${o.status}"\n`; }); const blob = new Blob([csv], {type: 'text/csv'}); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = `ghanahelp_orders_${new Date().toISOString().split('T')[0]}.csv`; a.click(); }
    </script>
