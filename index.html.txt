    <script>
        const services = ['Electrical', 'Plumbing', 'Cleaning', 'Renovation', 'IT Help', 'Telecoms'];
        const icons = ['⚡', '🔧', '🧹', '🔨', '💻', '📡'];
        let selected = [];
        function renderServices() { const grid = document.getElementById('serviceGrid'); grid.innerHTML = ''; services.forEach((s, i) => { const card = document.createElement('div'); card.className = 'service-card'; card.onclick = () => toggleService(s, card); card.innerHTML = `<div class="icon">${icons[i]}</div><h3>${s}</h3><div class="check-badge" id="check-${s}">✓ Selected</div>`; document.getElementById(`check-${s}`).style.display = 'none'; grid.appendChild(card); }); }
        function toggleService(s, card) { if (selected.includes(s)) { selected = selected.filter(x => x !== s); card.classList.remove('selected'); document.getElementById(`check-${s}`).style.display = 'none'; } else { selected.push(s); card.classList.add('selected'); document.getElementById(`check-${s}`).style.display = 'inline-block'; } updateForm(); }
        function updateForm() { const form = document.getElementById('formContainer'); const tags = document.getElementById('selectedTags'); form.classList.toggle('visible', selected.length > 0); tags.innerHTML = selected.map(s => `<span style="background:#E1F5EE; padding:5px 12px; border-radius:20px; margin:5px; display:inline-block;">✓ ${s}</span>`).join(''); if(selected.length > 0) setTimeout(() => form.scrollIntoView({behavior:'smooth'}), 100); }
        function submitOrder() { const name = document.getElementById('name').value.trim(); const phone = document.getElementById('phone').value.trim(); const location = document.getElementById('location').value.trim(); const desc = document.getElementById('description').value.trim(); if(!name || !phone || !location || !desc) { alert('Please fill all required fields'); return; } if(selected.length === 0) { alert('Please select a service'); return; }
            const order = { id: Date.now(), timestamp: new Date().toLocaleString(), name, phone, email: document.getElementById('email').value, location, services: selected.join(', '), urgency: document.getElementById('urgency').value, budget: document.getElementById('budget').value, description: desc, status: 'New' };
            let orders = JSON.parse(localStorage.getItem('ghanahelp_orders') || '[]'); orders.unshift(order); localStorage.setItem('ghanahelp_orders', JSON.stringify(orders));
            document.getElementById('successMsg').innerHTML = `Thanks ${name}! We'll contact you at ${phone} shortly.`; document.querySelector('.hero').style.display = 'none'; document.querySelector('.services').style.display = 'none'; document.getElementById('formContainer').classList.remove('visible'); document.getElementById('success').classList.add('visible'); window.scrollTo(0,0);
        }
        function resetForm() { selected = []; document.querySelectorAll('.service-card').forEach(c => c.classList.remove('selected')); document.querySelectorAll('.check-badge').forEach(b => b.style.display = 'none'); document.getElementById('name').value = ''; document.getElementById('phone').value = ''; document.getElementById('email').value = ''; document.getElementById('location').value = ''; document.getElementById('description').value = ''; document.getElementById('urgency').selectedIndex = 0; document.getElementById('budget').selectedIndex = 0; document.querySelector('.hero').style.display = ''; document.querySelector('.services').style.display = ''; document.getElementById('success').classList.remove('visible'); updateForm(); window.scrollTo(0,0); }
        renderServices();
    </script>
